### 첫번째 풀이
```cpp
#include <iostream>
#include <algorithm>
using namespace std;

string pel;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	cin >> pel;

	string tmp = pel;
	reverse(pel.begin(), pel.end());
	if (tmp == pel)
		cout << 1;
	else
		cout << 0;

	return 0;
}
```

### 두번째 풀이
```cpp
#include <iostream>
#include <stack>
using namespace std;

bool IsPalindrome(string s)
{
    int len = s.size();
    stack<char> st;
    int x, y;
    x = len / 2;
    if (len % 2 == 0) y = x;
    else y = x + 1;

    for (int i = 0; i < x; ++i)
    {
        st.push(s[i]);
    }
    for (int i = y; i < len; ++i)
    {
        if (st.top() == s[i])
        {
            st.pop();
        }
        else
        {
            return 0;
        }
    }

    return 1;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    string s;
    cin >> s;

    cout << IsPalindrome(s);

    return 0;
}
```
