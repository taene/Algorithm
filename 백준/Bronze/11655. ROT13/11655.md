### 최신 풀이
```cpp
#include <iostream>
#include <string>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    string S;
    getline(cin, S);

    for (int i = 0; i < S.size(); ++i)
    {
        if (S[i] == ' ' || S[i] < 'A' || S[i] > 'z') continue;

        if (S[i] >= 'a' && S[i] <= 'z')
        {
            S[i] = static_cast<char>((static_cast<int>(S[i] - 'a') + 13) % 26 + 'a');
        }
        else
        {
            S[i] = static_cast<char>((static_cast<int>(S[i] - 'A') + 13) % 26 + 'A');
        }
    }

    cout << S;

    return 0;
}
```
