```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int a[9];

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	for (int i = 0; i < 9; i++)
		cin >> a[i];

	int size = sizeof(a) / sizeof(a[0]);
	sort(a, a + size);

	do
	{
		int sum = 0;
		for (int i = 0; i < 7; i++)
			sum += a[i];

		if (sum == 100)
			break;
	} while (next_permutation(a, a + size));

	for (int i = 0; i < 7; i++)
		cout << a[i] << '\n';

	return 0;
}
```

### 두번째 풀이
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);

    int sum = 0;
    vector<int> v;
    for (int i = 0; i < 9; ++i)
    {
        int n;
        cin >> n;
        v.push_back(n);
        sum += n;
    }

    for (int i = 0; i < v.size(); ++i)
    {
        for (int j = 0; j < i; ++j)
        {
            if (v[i] + v[j] == sum - 100)
            {
                v[i] = -1;
                v[j] = -1;
                sort(v.begin(), v.end());
                for (int& it : v)
                {
                    if (it != -1)
                    {
                        cout << it << '\n';
                    }
                }

                return 0;
            }
        }
    }

    return 0;
}
```
