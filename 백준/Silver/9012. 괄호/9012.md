#include <iostream>
#include <stack>

using namespace std;

bool checkVPS(string& s)
{
    stack<char> stk;
    for (int i = 0; i < s.size(); ++i)
    {
        if (s[i] == '(')
        {
            stk.push(s[i]);
        }
        else
        {
            if (!stk.empty())
                stk.pop();
            else
            {
                return false;
            }
        }
    }
    
    return stk.empty();
}

int main(int argc, char* argv[])
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    string s;
    int t;
    cin >> t;
    while (t--)
    {
        cin >> s;

        if (checkVPS(s))
            cout << "YES" << '\n';
        else
            cout << "NO" << '\n';
    }

    return 0;
}
