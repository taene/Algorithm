### 최근 풀이
```cpp
#include <iostream>
using namespace std;

int N,M,K,Y,X;
int arr[51][51];
bool visited[51][51];
int dy[]={-1,0,1,0};
int dx[]={0,1,0,-1};

void dfs(int y, int x)
{
    visited[y][x]=true;
    for(int i=0; i<4; ++i)
    {
        int ny = y+dy[i];
        int nx = x+dx[i];
        if(ny<0||ny>=N||nx<0||nx>=M) continue;
        if(visited[ny][nx] || !arr[ny][nx]) continue;
        dfs(ny,nx);
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.tie(nullptr);
    
    int T;
    cin>>T;
    while(T--)
    {
        int ret = 0;
        fill(&visited[0][0],&visited[50][50], 0);
        fill(&arr[0][0], &arr[50][50], 0);
        cin>>M>>N>>K;
        for(int i=0; i<K; ++i)
        {
            cin>>X>>Y;
            arr[Y][X]=1;
        }
        
        for(int i=0; i<N; ++i)
        {
            for(int j=0; j<M; ++j)
            {
                if(arr[i][j] && !visited[i][j])
                {
                    ret++;
                    dfs(i,j);
                }
            }
        }

        cout<<ret<<'\n';
    }
    
    return 0;
}
```
