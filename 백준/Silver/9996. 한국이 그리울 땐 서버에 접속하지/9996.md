### 첫번째 풀이
```cpp
#include <iostream>
#include <vector>
using namespace std;

vector<string> split(string s, string del)
{
    vector<string> ret;
    int end = s.find(del);
    int start = 0;
    while (end != string::npos)
    {
        ret.push_back(s.substr(start, end - start));
        start = end + del.size();
        end = s.find(del, start);
    }
    ret.push_back(s.substr(start));
    return ret;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    string s;
    cin >> N >> s;

    vector<string> vec = split(s, "*");

    for (int i = 0; i < N; ++i)
    {
        bool check = false;
        string temp;
        cin >> temp;

        int findIdx = 0;
        for (int j = 0; j < vec.size(); ++j)
        {
            int idx = temp.find(vec[j], findIdx);
            if (idx == string::npos)
            {
                check = false;
                break;
            }
            else
            {
                check = true;
                findIdx = idx + vec[j].size();
            }
        }

        if (check)
        {
            cout << "DA" << '\n';
        }
        else
        {
            cout << "NE" << '\n';
        }
    }

    return 0;
}
```

<br/>

### 틀린 이유
- 처음에 *의 앞과 뒤의 문자열을 순서대로 찾아야하고 중복성을 고려하지 않았음. -> 이걸 *의 앞 문자열을 찾았을때 그 위치의 뒤부터 *의 뒤 문자열을 찾는것으로 해결함
- 근데 처음부터 문제를 잘못 해석함.. 입력받는 여러 문자열들의 처음과 끝이 *의 앞뒤 패턴 문자열로 이뤄져있어야 했음.
- 반례 또한 생각해야한다. (처음부터 *의 앞뒤 문자열의 크기 합과 입력받는 문자열의 크기를 비교해야 Out_of_range 오류가 안남)

### 최종 풀이
```cpp
#include <iostream>
#include <vector>
using namespace std;

vector<string> split(string s, string del)
{
    vector<string> ret;
    int end = s.find(del);
    int start = 0;
    while (end != string::npos)
    {
        ret.push_back(s.substr(start, end - start));
        start = end + del.size();
        end = s.find(del, start);
    }
    ret.push_back(s.substr(start));
    return ret;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    string s;
    cin >> N >> s;

    vector<string> vec = split(s, "*");

    for (int i = 0; i < N; ++i)
    {
        string temp;
        cin >> temp;

        if (vec[0].size() + vec[1].size() > temp.size())
        {
            cout << "NE" << '\n';
            continue;
        }

        bool check = (temp.substr(0, vec[0].size()) == vec[0]
            && temp.substr(temp.size() - vec[1].size()) == vec[1]);

        if (check) cout << "DA" << '\n';
        else cout << "NE" << '\n';
    }

    return 0;
}
```
