### 첫번째 풀이 (cnt)
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int n, k, A, cnt;
int weight = 500;
bool visited[10];
vector<int> v;
vector<int> ret;

void dfs()
{
    if (ret.size() == n)
    {
        cnt++;
        return;
    }

    for (int i = 0; i < n; ++i)
    {
        if (!visited[i])
        {
            visited[i] = true;
            ret.push_back(v[i]);
            weight += (v[i] - k);
            if (weight >= 500)
            {
                dfs();
            }
            weight -= (v[i] - k);
            ret.pop_back();
            visited[i] = false;
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    cin >> n >> k;
    for (int i = 0; i < n; ++i)
    {
        cin >> A;
        v.push_back(A);
    }
    dfs();
    cout << cnt;

    return 0;
}
```

### 두번째 풀이 (int dfs)
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int n, k, A;
int weight = 500;
bool visited[10];
vector<int> v;
vector<int> ret;

int dfs()
{
    int token = 0;

    if (ret.size() == n)
    {
        return 1;
    }

    for (int i = 0; i < n; ++i)
    {
        if (!visited[i])
        {
            visited[i] = true;
            ret.push_back(v[i]);
            weight += (v[i] - k);
            if (weight >= 500)
            {
                token += dfs();
            }
            weight -= (v[i] - k);
            ret.pop_back();
            visited[i] = false;
        }
    }

    return token;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    cin >> n >> k;
    for (int i = 0; i < n; ++i)
    {
        cin >> A;
        v.push_back(A);
    }
    cout << dfs();

    return 0;
}
```
