### 첫번째 풀이
```cpp
#include <iostream>
#include <map>
#include <algorithm>
#include <string>

using namespace std;

int main(int argc, char* argv[])
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    int n, m;
    map<int, string> mp1;
    map<string, int> mp2;

    cin >> n >> m;
    for (int i = 1; i <= n; ++i)
    {
        string temp;
        cin >> temp;
        mp1[i] = temp;
        mp2[temp] = i;
    }
    for (int i = 0; i < m; ++i)
    {
        string temp;
        cin >> temp;
        if (temp[0] >= 'A' && temp[0] <= 'Z')
        {
            cout << mp2[temp] << '\n';
        }
        else
        {
            cout << mp1[stoi(temp)] << '\n';
        }
    }

    return 0;
}
```

### 두번째 풀이
```cpp
#include <iostream>
#include <map>

using namespace std;

int main(int argc, char* argv[])
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    int n, m;
    map<int, string> mp1;
    map<string, int> mp2;

    cin >> n >> m;
    for (int i = 1; i <= n; ++i)
    {
        string temp;
        cin >> temp;
        mp1[i] = temp;
        mp2[temp] = i;
    }
    for (int i = 0; i < m; ++i)
    {
        string temp;
        cin >> temp;
        if (int num = atoi(temp.c_str()))
        {
            cout << mp1[num] << '\n';
        }
        else
        {
            cout << mp2[temp] << '\n';
        }
    }

    return 0;
}
```
