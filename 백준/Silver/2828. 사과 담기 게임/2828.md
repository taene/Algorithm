### 최근 풀이
```cpp
#include <iostream>
using namespace std;

int N,M,J;
int ret;
int startIdx, endIdx;

int main()
{
    cin>>N>>M>>J;
    startIdx = 1;
    endIdx = M;
    for(int i=0; i<J; ++i)
    {
        int loc;
        cin>>loc;
        
        if(loc<startIdx)
        {
            ret+=startIdx-loc;
            startIdx=loc;
            endIdx=loc+(M-1);
        }
        else if(loc>endIdx)
        {
            ret+=loc-endIdx;
            endIdx=loc;
            startIdx=loc-(M-1);
        }
    }
    cout<<ret;
    
    return 0;
}
```
