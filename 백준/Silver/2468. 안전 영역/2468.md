### 최근 풀이
```cpp
#include <iostream>
using namespace std;

int N, mx;
int mn = 101;
int arr[101][101];
bool visited[101][101];
int dy[]={-1,0,1,0};
int dx[]={0,1,0,-1};

void dfs(int y, int x, int height)
{
    visited[y][x] = true;
    
    for(int i=0; i<4; ++i)
    {
        int ny = y+dy[i];
        int nx = x+dx[i];
        
        if(ny<0||ny>=N||nx<0||nx>=N) continue;
        if(visited[ny][nx]) continue;
        if(arr[ny][nx] <= height) continue;

        dfs(ny,nx,height);
    }
}

int main()
{
    cin>>N;
    for(int i=0; i<N; ++i)
    {
        for(int j=0; j<N; ++j)
        {
            cin>>arr[i][j];
            mx = max(arr[i][j], mx);
            mn = min(arr[i][j], mn);
        }
    }
    
    int ret = 1;
    for(int k=mn; k<mx; ++k)
    {
        int temp = 0;
        fill(&visited[0][0], &visited[100][100], 0);
        for(int i=0; i<N; ++i)
        {
            for(int j=0; j<N; ++j)
            {
                if(!visited[i][j] && arr[i][j]>k)
                {
                    temp++;
                    dfs(i,j,k);
                }
            }
        }
        ret = max(ret,temp);
    }
    cout<<ret;
    
    return 0;
}
```
