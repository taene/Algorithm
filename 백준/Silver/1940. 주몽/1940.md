### 최종 풀이
```cpp
#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int N, M;
    vector<int> v;
    cin >> N >> M;
    for (int i = 0; i < N; ++i)
    {
        int tmp;
        cin >> tmp;
        v.push_back(tmp);
    }

    int ret = 0;
    for (int i = 0; i < N; ++i)
    {
        for (int j = 0; j < i; ++j)
        {
            if (v[i] + v[j] == M)
            {
                ret++;
            }
        }
    }
    cout << ret;

    return 0;
}
```
- N의 최대 범위가 15,000 이기 때문에 15,000C2(Combi) = 약 1억 2750만 으로 2초의 시간 제한 조건을 충족하여, 알고리즘 중 조합을 써도 된다.
