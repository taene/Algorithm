### 최근 풀이
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int x_1,y_1,x_2,y_2;
int M,N,K,ret;
int arr[101][101];
bool visited[101][101];
int dy[]={-1,0,1,0};
int dx[]={0,1,0,-1};
vector<int> ext;

int dfs(int y, int x)
{
    int token = 1;
    visited[y][x] = true;
    
    for(int i=0; i<4; ++i)
    {
        int ny = y+dy[i];
        int nx = x+dx[i];
        
        if(ny<0||ny>=M||nx<0||nx>=N) continue;
        if(visited[ny][nx]) continue;
        if(arr[ny][nx]) continue;
        
        token += dfs(ny,nx);
    }
    
    return token;
}

int main()
{
    cin>>M>>N>>K;
    for(int i=0; i<K; ++i)
    {
        cin>>x_1>>y_1;
        cin>>x_2>>y_2;
        
        for(int x=x_1; x<x_2; ++x)
        {
            for(int y=y_1; y<y_2; ++y)
            {
                arr[y][x] = 1;
            }
        }
    }
    
    for(int i=0; i<M; ++i)
    {
        for(int j=0; j<N; ++j)
        {
            if(arr[i][j]==0 && !visited[i][j])
            {
                ret++;
                ext.push_back(dfs(i,j));
            }
        }
    }
    
    sort(ext.begin(), ext.end());
    
    cout<<ret<<'\n';
    for(int it:ext)
    {
        cout<<it<<' ';
    }
    
    return 0;
}
```
