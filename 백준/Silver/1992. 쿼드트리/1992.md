### 최근 풀이
```cpp
#include <iostream>
#include <vector>
using namespace std;

int N;
vector<int> v[65];

void dfs(int y, int x, int n)
{
    for(int i=y; i<y+n; ++i)
    {
        for(int j=x; j<x+n; ++j)
        {
            if(v[y][x] != v[i][j])
            {
                n/=2;
                
                cout<<"(";
                dfs(y,x,n);
                dfs(y,x+n,n);
                dfs(y+n,x,n);
                dfs(y+n,x+n,n);
                cout<<")";
                
                return;
            }
        }
    }
    
    cout<<v[y][x];
}

int main()
{
    cin>>N;
    for(int i=0; i<N; ++i)
    {
        string s;
        cin>>s;
        for(int j=0; j<N; ++j)
        {
            v[i].push_back(static_cast<int>(s[j]-'0'));
        }
    }
    
    dfs(0,0,N);
    
    return 0;
}
```
