### 풀이 1
```cpp
#include <iostream>
using namespace std;

string name, result;
int cnt[26];
int oddNum;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	cin >> name;

	for (auto i : name)
		cnt[i - 'A']++;

	for (int i = 0; i < 26; i++)
	{
		// 홀수표현: cnt[i] & 1 (비트연산자)
		if (cnt[i] && cnt[i] % 2 == 1)
		{
			result += (i + 'A');
			oddNum++;
		}
	}

	if (oddNum >= 2)
		cout << "I'm Sorry Hansoo" << '\n';
	else
	{
		for (int i = 25; i >= 0; i--)
		{
			if (cnt[i])
			{
				int half = cnt[i] / 2;
				string temp(half, char(i + 'A'));
				result = temp + result + temp;
			}
		}

		cout << result << '\n';
	}

	return 0;
}
```

### 풀이 2
```cpp
#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int alphabet[26] = {};
    string s;
    cin >> s;
    for (int i = 0; i < s.size(); ++i)
    {
        alphabet[static_cast<int>(s[i] - 'A')]++;
    }

    int oddCnt = 0;
    int oddIdx = -1;
    for (int i = 0; i < 26; ++i)
    {
        if (alphabet[i] != 0 && alphabet[i] % 2 == 1)
        {
            oddCnt++;
            if (oddCnt > 1)
            {
                cout << "I'm Sorry Hansoo";
                return 0;
            }
            oddIdx = i;
        }
    }

    string ret = "";
    if (oddIdx != -1)
    {
        ret += static_cast<char>(oddIdx + 'A');
        alphabet[oddIdx]--;
    }

    string prefix = "";
    for (int i = 25; i >= 0; --i)
    {
        if (alphabet[i])
        {
            for (int j = 0; j < alphabet[i] / 2; ++j)
            {
                prefix += static_cast<char>(i + 'A');
            }
        }
    }

    string suffix = prefix;
    reverse(prefix.begin(), prefix.end());
    ret = prefix + ret + suffix;
    cout << ret;

    return 0;
}
```
- reverse()는 <algorithm> 헤더를 써야함
