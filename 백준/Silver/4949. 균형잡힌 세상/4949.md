#include <iostream>
#include <string>
#include <stack>

using namespace std;

bool checkString(string& s)
{
    stack<char> stk;

    for (int i = 0; i < s.size(); ++i)
    {
        if (s[i] == '(' || s[i] == '[')
        {
            stk.push(s[i]);
        }
        else if (s[i] == ')')
        {
            if (!stk.empty() && stk.top() == '(')
            {
                stk.pop();
            }
            else
            {
                return false;
            }
        }
        else if (s[i] == ']')
        {
            if (!stk.empty() && stk.top() == '[')
            {
                stk.pop();
            }
            else
            {
                return false;
            }
        }
    }

    return stk.empty();
}

int main(int argc, char* argv[])
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    string s;
    while (true)
    {
        getline(cin, s);

        if (s[0] == '.' && s.size() == 1)
            break;

        if (checkString(s))
        {
            cout << "yes" << '\n';
        }
        else
        {
            cout << "no" << '\n';
        }
    }

    return 0;
}
