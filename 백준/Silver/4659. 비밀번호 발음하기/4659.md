### 최근 풀이
```cpp
#include <iostream>
using namespace std;

void Print(bool b, string s)
{
    if(b)
    {
        cout<<"<"<<s<<"> is acceptable.\n";
    }
    else
    {
        cout<<"<"<<s<<"> is not acceptable.\n";
    }
}

bool IsValid_AEIOU(string s)
{
    for(char it : s)
    {
        if(it=='a' || it=='e' || it=='i'
            || it=='o'|| it=='u')
        {
            return true;
        }
    }
    return false;
}

bool IsValid_Triple(string s)
{
    int cntA=0;
    int cntB=0;
    for(int i=0; i<s.size(); ++i)
    {
        if(cntA==3 || cntB==3)
        {
            return false;
        }
        
        if(IsValid_AEIOU(string(1,s[i])))
        {
            cntA++;
            cntB=0;
        }
        else
        {
            cntB++;
            cntA=0;
        }
    }
    
    if(cntA==3 || cntB==3)
    {
        return false;
    }
    
    return true;
}

bool IsValid_SameAlphabet(string s)
{
    for(int i=1; i<s.size(); ++i)
    {
        if(s[i]==s[i-1] 
            && s[i]!='e' && s[i]!='o')
        {
            return false;
        }
    }
    return true;
}

int main()
{
    // aeiou 하나 포함
    // 모음3개연속or자음3개연속 X
    // 같은글자 2번연속X, [ee,oo는 됨]
    
    string s;
    while(getline(cin, s))
    {
        if(s=="end")
        {
            return 0;
        }
        
        Print(IsValid_AEIOU(s)
            && IsValid_Triple(s)
            && IsValid_SameAlphabet(s), s);
    }
    
    return 0;
}
```
