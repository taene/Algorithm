#include <map>

using namespace std;

int main(int argc, char* argv[])
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);

    int n, m;
    map<int, int> mp;
    cin >> n;
    for (int i = 0; i < n; ++i)
    {
        int temp;
        cin >> temp;
        mp[temp]++;
    }

    cin >> m;
    for (int i = 0; i < m; ++i)
    {
        int temp;
        cin >> temp;
        cout << mp[temp] << ' ';
    }

    return 0;
}
